HTTP Client Request Testing

This is a dummy site created by Mysterious Trousers (hosted on Heroku) so that anyone can test any HTTP request frameworks they are working on.

Formats

The site will respond to HTML, XML and JSON requests.

Resources

There are two resources: stitches and needes. The needles resources needs http basic authentication. (username: "username", password: "password").

    create_table "needles", :force => true do |t|
      t.integer  "sharpness"
      t.integer  "length"
      t.datetime "created_at"
      t.datetime "updated_at"
    end

    create_table "stitches", :force => true do |t|
      t.string   "thread_color"
      t.string   "length"
      t.datetime "created_at"
      t.datetime "updated_at"
    end

Here are the REST routes you can hit: 

          needles GET    /needles(.:format)           {:action=>"index", :controller=>"needles"}
                  POST   /needles(.:format)           {:action=>"create", :controller=>"needles"}
       new_needle GET    /needles/new(.:format)       {:action=>"new", :controller=>"needles"}
      edit_needle GET    /needles/:id/edit(.:format)  {:action=>"edit", :controller=>"needles"}
           needle GET    /needles/:id(.:format)       {:action=>"show", :controller=>"needles"}
                  PUT    /needles/:id(.:format)       {:action=>"update", :controller=>"needles"}
                  DELETE /needles/:id(.:format)       {:action=>"destroy", :controller=>"needles"}
         stitches GET    /stitches(.:format)          {:action=>"index", :controller=>"stitches"}
                  POST   /stitches(.:format)          {:action=>"create", :controller=>"stitches"}
       new_stitch GET    /stitches/new(.:format)      {:action=>"new", :controller=>"stitches"}
      edit_stitch GET    /stitches/:id/edit(.:format) {:action=>"edit", :controller=>"stitches"}
           stitch GET    /stitches/:id(.:format)      {:action=>"show", :controller=>"stitches"}
                  PUT    /stitches/:id(.:format)      {:action=>"update", :controller=>"stitches"}
                  DELETE /stitches/:id(.:format)      {:action=>"destroy", :controller=>"stitches"}
